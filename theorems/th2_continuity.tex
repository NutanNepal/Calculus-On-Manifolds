\begin{theorem}
    \label{topological continuity}
    If $A\subset \ron$, a function $f:A\to \rom$
    is continuous if and only if for every open
    set $U\subset\rom$ there is some open set
    $V\subset\ron$ such that $f^{-1}(U)=V\cap A$.
\end{theorem}
