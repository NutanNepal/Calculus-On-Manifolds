\begin{theorem}[Heine-Borel Theorem]
    \label{heineborel}
    The closed interval $[a,b]$ is compact.
\end{theorem}

\begin{tcolorbox}
    \begin{proof}
        If $\mathscr{O}$ is an open cover of $[a,b]$, let
        $$A=\{x: a\le x\le b \text{ and $[a,x]$ is covered
        by some finite number of open sets in $\mathscr{O}$}\}$$
    \end{proof}
\end{tcolorbox}
