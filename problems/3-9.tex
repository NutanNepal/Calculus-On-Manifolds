\question{
    [3-9]
    
    Show that an unbounded set cannot have
    content 0.
    
    Give an example of a closed set of
    measure 0 which does not have
    content 0.
}
\begin{solution}
    \begin{parts}
        \part{
    Unbounded sets cannot have finite cover
    and hence do not have content 0.
        }
        \part{
    $\mathbb{Z}$ has measure 0 but not
    content 0.
        }
    \end{parts}


\end{solution}
